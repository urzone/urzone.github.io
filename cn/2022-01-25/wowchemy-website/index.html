  • 选用各部分字体组成一个字体集： <WebsiteDir>/data/fonts/local_font.toml
# Font style metadata
name = "Local_Font"

# Optional Google font URL
google_fonts = ""

# Font families
heading_font = "EB Garamond"
body_font = "EB Garamond"
nav_font = "EB Garamond"
mono_font = "EB Garamond Mono"
  • 将选用的字体所构成的字体集引入网页：<WebsiteDir>/config/_default/params.toml
# Enable users to switch between day and night mode?
day_night: true

# Change the "font" value to 'native' to use native fonts optimized for the user’s device
# Override the theme's font set (optional).
#   Latest font sets (may require updating): https://sourcethemes.com/academic/themes/
#   Browse built-in font sets in `themes/academic/data/fonts/`
#   Browse user installed font sets in `data/fonts/`
font: "Local_Font" #Local_Font

# Choose a font size.
# Sizes: XS (extra small), S (small), M (medium), L (large - DEFAULT), XL (extra large)
font_size = "L"
